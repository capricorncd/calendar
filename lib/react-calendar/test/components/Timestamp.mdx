```jsx
import React, { Component } from 'react'
import { ZxReactCalendar } from '../../index'

class App extends Component {
  constructor(props) {
    super(props)
    this.state = {
      date: +new Date('1999/09/09')
    }
  }

  handleChange(res, original) {
    this.setState({
      date: res
    })
  }

  render() {
    return <>
      <h2>format: timestamp</h2>
      <p>value: {this.state.date}</p>
      <div className="example-demo-wrapper">
        <ZxReactCalendar
          lang="jp"
          format="timestamp"
          value={this.state.date}
          change={(res, original) => { this.handleChange(res, original) }}
        />
        INJECT_CODE
      </div>
    </>
  }
}

export default App
```
